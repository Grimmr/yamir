root = data root?
     = function root?
data = DAT W NUM 
function = fn_name (W NUM)? L_PAREN param_list? R_PARAM L_BRACE body? R_BRACE
fn_name = HASH IDENT
param_list = param COMA param_list?
param = var W NUM
body = line body?
line = label
     = instruction
     = call
label = COLON IDENT COLON
instruction = OP v_list
v_list = v v_list?
v = immediate
  = var
  = pointer_var
  = pointer_immediate
  = label_val
  = call
immediate = BANG NUM 
var = DOLLAR IDENT
pointer_var = AT IDENT 
pointer  = AMP NUM 
label_val  = COLON IDENT 
call = fn_name IDENT L_PAREN call_list? R_PAREN
call_list = v COMA v?
